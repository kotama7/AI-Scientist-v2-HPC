                try:
                    candidate_paths = []
                    # Prefer *_data.npy pattern (e.g., {experiment_name}_data.npy)
                    candidate_paths += glob.glob(os.path.join(working_dir, '*_data.npy'))
                    candidate_paths += glob.glob(os.path.join(working_dir, 'experiment_data_*.npy'))
                    candidate_paths += glob.glob(os.path.join(working_dir, 'output*.npy'))
                    if not candidate_paths:
                        candidate_paths = glob.glob(os.path.join(working_dir, '**', '*.npy'), recursive=True)
                    candidate_paths = sorted(set(candidate_paths))

                    all_experiment_data = []
                    for p in candidate_paths:
                        if not os.path.isfile(p):
                            print(f'Warning: experiment data file not found, skipping: {p}')
                            continue
                        try:
                            loaded = np.load(p, allow_pickle=True)
                            experiment_data = loaded.item() if isinstance(loaded, np.ndarray) and loaded.shape == () else loaded
                            all_experiment_data.append((p, experiment_data))
                        except Exception as e:
                            print(f'Error loading experiment data {p}: {e}')
                except Exception as e:
                    print(f'Error loading experiment data: {e}')
