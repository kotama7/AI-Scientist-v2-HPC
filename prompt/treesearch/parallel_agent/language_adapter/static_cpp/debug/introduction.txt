#include <cnpy.h>
#include <filesystem>
#include <iostream>
#include <vector>
#include <cmath>
namespace fs = std::filesystem;

fs::path resolve_working_dir() {
    fs::path script_path = fs::weakly_canonical(fs::path(__FILE__));
    fs::path base_dir = script_path.has_parent_path() ? script_path.parent_path() : fs::current_path();
    fs::path working_dir = base_dir / "working";
    fs::create_directories(working_dir);
    return working_dir;
}

void saveExperimentData(const fs::path& working_dir, const std::vector<double>& data) {
    const fs::path target = working_dir / "experiment_data.npy";
    cnpy::npy_save(target.string(), data.data(), {data.size()}, "w");
    if (!fs::exists(target)) {
        throw std::runtime_error("experiment_data.npy was not written correctly.");
    }
}

std::vector<double> loadExperimentData(const fs::path& working_dir) {
    const fs::path target = working_dir / "experiment_data.npy";
    cnpy::NpyArray arr = cnpy::npy_load(target.string());
    auto* loaded = arr.data<double>();
    return std::vector<double>(loaded, loaded + arr.num_bytes() / sizeof(double));
}

void logSummary(const std::vector<double>& data) {
    double sum = 0.0;
    for (double value : data) {
        sum += value;
    }
    double mean = data.empty() ? 0.0 : sum / data.size();
    std::cout << "Data points: " << data.size() << ", Mean: " << mean << std::endl;
}

int main() {
    fs::path working_dir = resolve_working_dir();

    // Placeholder for experiment data generation
    std::vector<double> experimentData = {/* experiment data */};

    // Save the experiment data
    saveExperimentData(working_dir, experimentData);

    // Load the experiment data
    std::vector<double> loadedData = loadExperimentData(working_dir);

    logSummary(loadedData);

    return 0;
}
