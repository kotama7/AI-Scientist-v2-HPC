#include <iostream>
#include <filesystem>
#include <fstream>
#include <vector>
#include <string>
#include <cnpy.h>

void loadExperimentData(const std::string& filePath, std::vector<std::vector<double>>& data) {
    // Load the .npy file into the data vector
    cnpy::NpyArray arr = cnpy::npy_load(filePath);
    auto* loaded = arr.data<double>();
    if (!loaded) {
        throw std::runtime_error("Failed to load experiment data with cnpy");
    }
    std::vector<double> flat(loaded, loaded + arr.num_bytes() / sizeof(double));
    data.push_back(flat);
}

void printMetrics(const std::vector<std::vector<double>>& data) {
    for (const auto& dataset : data) {
        std::cout << "Dataset Name: " << /* Extract dataset name */ << std::endl;
        for (const auto& metric : dataset) {
            std::cout << "Metric Name: " << /* Extract metric name */ << " Value: " << metric << std::endl;
        }
    }
}

std::filesystem::path resolveWorkingDir() {
    std::filesystem::path scriptPath = std::filesystem::weakly_canonical(std::filesystem::path(__FILE__));
    std::filesystem::path baseDir = scriptPath.has_parent_path() ? scriptPath.parent_path() : std::filesystem::current_path();
    std::filesystem::path workingDir = baseDir / "working";
    std::filesystem::create_directories(workingDir);
    return workingDir;
}

int main() {
    std::filesystem::path workingDir = resolveWorkingDir();
    std::string filePath = (workingDir / "experiment_data.npy").string();
    
    std::vector<std::vector<double>> experimentData;
    loadExperimentData(filePath, experimentData);
    
    printMetrics(experimentData);
    
    return 0;
}
