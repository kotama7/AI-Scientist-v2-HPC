#include <iostream>
#include <fstream>
#include <vector>
#include <cmath>
#include <matplotlibcpp.h>

namespace plt = matplotlibcpp;

void load_data(const std::string& filename, std::vector<double>& data) {
    std::ifstream file(filename, std::ios::binary);
    if (!file) {
        std::cerr << "Error opening file: " << filename << std::endl;
        return;
    }
    double value;
    while (file.read(reinterpret_cast<char*>(&value), sizeof(double))) {
        data.push_back(value);
    }
    file.close();
}

void save_data(const std::string& filename, const std::vector<double>& data) {
    std::ofstream file(filename, std::ios::binary);
    if (!file) {
        std::cerr << "Error opening file: " << filename << std::endl;
        return;
    }
    for (const double& value : data) {
        file.write(reinterpret_cast<const char*>(&value), sizeof(double));
    }
    file.close();
}

double compute_metric(const std::vector<double>& data) {
    double sum = 0.0;
    for (const double& value : data) {
        sum += value;
    }
    return sum / data.size();
}

int main() {
    std::vector<double> experiment_data;
    load_data("experiment_data.npy", experiment_data);

    double metric = compute_metric(experiment_data);
    std::cout << "Evaluation Metric: " << metric << std::endl;

    plt::plot(experiment_data);
    plt::show();

    save_data("experiment_data.npy", experiment_data);

    return 0;
}
