```cpp
#include <iostream>
#include <fstream>
#include <vector>
#include <cmath>
#include <random>
#include <matplotlibcpp.h>

namespace plt = matplotlibcpp;

std::vector<double> load_data(const std::string& filename) {
    std::ifstream file(filename, std::ios::binary);
    std::vector<double> data;
    double value;
    while (file.read(reinterpret_cast<char*>(&value), sizeof(double))) {
        data.push_back(value);
    }
    return data;
}

void save_data(const std::vector<double>& data, const std::string& filename) {
    std::ofstream file(filename, std::ios::binary);
    for (double value : data) {
        file.write(reinterpret_cast<const char*>(&value), sizeof(double));
    }
}

std::vector<double> generate_data(int size, double mean, double stddev) {
    std::random_device rd;
    std::mt19937 gen(rd());
    std::normal_distribution<> d(mean, stddev);

    std::vector<double> data(size);
    for (int i = 0; i < size; ++i) {
        data[i] = d(gen);
    }
    return data;
}

void plot_data(const std::vector<double>& data) {
    plt::plot(data);
    plt::show();
}

int main() {
    const std::string filename = "experiment_data.npy";
    const int data_size = 1000;
    const double mean = 0.0;
    const double stddev = 1.0;

    std::vector<double> data = generate_data(data_size, mean, stddev);
    save_data(data, filename);

    std::vector<double> loaded_data = load_data(filename);
    plot_data(loaded_data);

    return 0;
}
```
